function modEfectivo(){ return modulo * (1 + markup/100); }

function fmt(n){ return '$\u00a0' + Math.round(n).toLocaleString('es-AR'); }

function fmtM(n){ return n % 1 === 0 ? ''+n : n.toFixed(1); }


function getMods(item, s){

  if(!s || !s.on) return 0;

  const c = s.comp || 'm', qty = parseFloat(s.qty)||1;

  if(item.fixed !== undefined) return item.fixed * qty;

  const base = item[c] || 0;

  if(item.byHour) return Math.max(base * qty, 30);

  return base * qty;

}

function renderSection(sec){

  const el = document.getElementById('section-'+sec);

  const isT = sec==='t';

  el.innerHTML = DATA[sec].map(g => {

    const isOpen = openGroups[g.gid];

    const rows = g.items.map((item,idx) => {

      const s = sel[item.id] || {on:false, comp:'m', qty:1};

      const mods = getMods(item, s);

      const me = modEfectivo();

      const hasComp = item.fixed===undefined && !item.byHour;

      const isHour = item.byHour;

      const cls = s.on ? (isT?'checked-t':'checked-l') : (idx%2===0?'even':'odd');

      const compSel = (hasComp||isHour)

        ? `<select onchange="updateSel('${item.id}','comp',this.value)" ${s.on?'':'disabled'}>

            <option value="a" ${(s.comp||'m')==='a'?'selected':''}>Alta</option>

            <option value="m" ${(s.comp||'m')==='m'?'selected':''}>Media</option>

            <option value="b" ${(s.comp||'m')==='b'?'selected':''}>Baja</option>

           </select>`

        : '<span style="font-size:11px;color:#bbb">—</span>';

      const qtyEl = `<div style="display:flex;align-items:center;gap:3px">

        <input type="number" min="1" value="${s.qty||1}" ${s.on?'':'disabled'}

          style="width:46px;text-align:center"

          onchange="updateSel('${item.id}','qty',Math.max(1,parseInt(this.value)||1))">

        <span style="font-size:10px;color:#aaa">${isHour?'hs':'×'}</span>

      </div>`;

      const modsEl = `<div class="mods ${s.on&&mods>0?'on':''}">${s.on&&mods>0?fmtM(mods):'—'}</div>`;

      const nameEl = `<div class="item-name ${s.on?'on':''}">${item.n}${item.fixed!==undefined?`<span class="fixed-badge"> (fijo: ${item.fixed})</span>`:''}</div>`;

function renderHoras(){

}


function updateTotals(){

  const me = modEfectivo();

  let tribM=0, labM=0;

  DATA.t.forEach(g=>g.items.forEach(it=>{ tribM+=getMods(it,sel[it.id]); }));

  DATA.l.forEach(g=>g.items.forEach(it=>{ labM+=getMods(it,sel[it.id]); }));

  const horasM = ROLES.reduce((a,r)=>a+(parseFloat(horas[r.id])||0)*r.modsH,0);

  const total = tribM+labM+horasM;


  document.getElementById('badge-t').textContent = tribM>0?fmt(tribM*me):'$0';

  document.getElementById('badge-l').textContent = labM>0?fmt(labM*me):'$0';

  document.getElementById('total-amount').textContent = fmt(total*me);

  document.getElementById('total-mods').textContent = fmtM(total)+' módulos totales';

  document.getElementById('total-sub').textContent =

    `Tributario: ${fmtM(tribM)} mód. · Laboral: ${fmtM(labM)} mód. · Horas: ${fmtM(horasM)} mód.`;


  const tmEl = document.getElementById('total-markup');

  if(markup>0){

    tmEl.style.display='block';

    tmEl.textContent=`Módulo efectivo con mark-up ${markup}%: ${fmt(me)}`;

  } else { tmEl.style.display='none'; }


  const mebar = document.getElementById('modulo-efectivo-bar');

  const mkbox = document.getElementById('markup-box');

  if(markup>0){

    mebar.style.display='block';

    mebar.innerHTML=`<strong>Módulo efectivo:</strong> ${fmt(modulo)} × ${(1+markup/100).toFixed(2)} = <strong>${fmt(me)}</strong> por módulo`;

    mkbox.classList.add('markup');

  } else {

    mebar.style.display='none';

    mkbox.classList.remove('markup');

  }

}

  updateTotals();

}


// ── EVENTOS ───────────────────────────────────────────────────────────

function toggleItem(id, checked){

  if(!sel[id]) sel[id]={on:false,comp:'m',qty:1};

  sel[id].on=checked;

  render();

}

function updateSel(id, key, val){

  if(!sel[id]) sel[id]={on:true,comp:'m',qty:1};

  sel[id][key]=val;

  render();

}

function updateHoras(id, val){

  horas[id]=parseFloat(val)||0;

  renderHoras();

  updateTotals();

}

function toggleGroup(gid){

  openGroups[gid]=!openGroups[gid];

  render();

}

function toggleGH(){

  ghOpen=!ghOpen;

  document.getElementById('body-gh').className='group-body'+(ghOpen?' open':'');

  document.getElementById('arrow-gh').textContent=ghOpen?'▲':'▼';

}

function switchTab(t){

  activeTab=t;

  document.getElementById('section-t').style.display=t==='t'?'':'none';

  document.getElementById('section-l').style.display=t==='l'?'':'none';

  document.getElementById('tab-t').className='tab'+(t==='t'?' active-t':'');

  document.getElementById('tab-l').className='tab'+(t==='l'?' active-l':'');

}

function onModulo(v){

  const n=parseFloat(v.replace(/\./g,'').replace(',','.'));

  if(!isNaN(n)&&n>0){ modulo=n; render(); }

}

function onMarkup(v){

  const n=parseFloat(v.replace(',','.'));

  if(!isNaN(n)&&n>=0){ markup=n; render(); }

}

function clearAll(){

  sel={};

  horas={jr:0,sr:0,socio:0};

  render();

}

function init(){

  [...DATA.t,...DATA.l].forEach(g=>{ openGroups[g.gid]=true; });

  render();

}
